101 const int MAX_CHAR = 26;
102 
103 void commonCharacters(string str[], int n)
104 {
105     // primary array for common characters  
106     // we assume all characters are seen before. 
107     bool prim[MAX_CHAR];
108     memset(prim, true, sizeof(prim));
109 
110     // for each string 
111     for (int i = 0; i < n; i++) {
112 
113         // secondary array for common characters 
114         // Initially marked false 
115         bool sec[MAX_CHAR] = { false };
116 
117         // for every character of ith string 
118         for (int j = 0; str[i][j]; j++) {
119 
120             // if character is present in all  
121             // strings before, mark it. 
122             if (prim[str[i][j] - 'a'])
123                 sec[str[i][j] - 'a'] = true;
124         }
125 
126         // copy whole secondary array into primary 
127         memcpy(prim, sec, MAX_CHAR);
128     }
129 
130     // displaying common characters 
131     for (int i = 0; i < 26; i++)
132         if (prim[i])
133             printf("%c ", i + 'a');
134 }
